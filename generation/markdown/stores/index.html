    Stores - Reactive State Management¶

    Stores are the foundation of reactive state management in FynX. They provide a structured way to organize related observables, computed properties, and reactive methods into cohesive units that represent different parts of your application's state.

    What are Stores?¶

    A Store is a class that groups related reactive state together. Stores combine the power of observables, computed properties, and reactive methods to create self-contained state management units.

    Key Benefits¶

    • Organization: Group related state in logical units
    • Encapsulation: Keep state logic contained within store classes
    • Reusability: Stores can be easily reused across different parts of your application
    • Testability: Isolated state makes testing straightforward
    • Type Safety: Full type hint support for better IDE experience

    Basic Store Usage¶

    Creating a Simple Store¶

    from fynx import Store, observable

class CounterStore(Store):
    # Observable state
    count = observable(0)
    step = observable(1)

    # Computed properties using the >> operator
    doubled_count = count >> (lambda c: c * 2)

    # Methods to modify state
    @classmethod
    def increment(cls):
        cls.count = cls.count.value + cls.step.value

    @classmethod
    def decrement(cls):
        cls.count = cls.count.value - cls.step.value

# Usage
CounterStore.increment()  # count becomes 1, doubled_count becomes 2
CounterStore.step = 5
CounterStore.increment()  # count becomes 6, doubled_count becomes 12

    Reactive Class Attributes¶

    Use the @observable decorator to make class attributes reactive:

    class UserStore(Store):
    name = observable("Anonymous")
    age = observable(25)
    is_online = observable(False)

# Changes trigger reactive updates throughout your application
UserStore.name = "Alice"
UserStore.age = 30
UserStore.is_online = True

    Advanced Store Patterns¶

    Complex Computed Properties¶

    Stores excel at managing complex derived state through computed properties:

    from fynx import Store, observable

class ShoppingCartStore(Store):
    items = observable([
        {"name": "Widget", "price": 10.99, "quantity": 2},
        {"name": "Gadget", "price": 15.50, "quantity": 1}
    ])

    # Basic computed properties using the >> operator
    total_items = items >> (lambda items_list: sum(item["quantity"] for item in items_list))
    subtotal = items >> (lambda items_list: sum(item["price"] * item["quantity"] for item in items_list))

    # Progressive tax rate based on subtotal (computed from subtotal)
    tax_rate = subtotal >> (lambda total:
        0.08 if total < 50 else  # 8% for small orders
        0.10 if total < 100 else  # 10% for medium orders
        0.12  # 12% for large orders
    )

    # Chained computed properties
    tax_amount = (subtotal | tax_rate) >> (lambda total, rate: total * rate)
    total_price = (subtotal | tax_amount) >> (lambda sub, tax: sub + tax)

    average_item_price = (subtotal | total_items) >> (
        lambda total, count: total / count if count > 0 else 0
    )

# Usage
cart = ShoppingCartStore()
print(f"Total: ${cart.total_price:.2f}")  # Total: $47.48

# Adding more expensive items changes the tax rate automatically
ShoppingCartStore.items = cart.items + [
    {"name": "Expensive Item", "price": 200, "quantity": 1}
]
print(f"New total: ${cart.total_price:.2f}")  # New total: $291.48 (higher tax rate)

    Filtering and Conditional Logic¶

    Use computed properties with the >> operator for complex filtering and transformations:

    class TodoStore(Store):
    todos = observable([
        {"id": "1", "text": "Learn FynX", "completed": False, "priority": "high"},
        {"id": "2", "text": "Build app", "completed": True, "priority": "medium"},
        {"id": "3", "text": "Write tests", "completed": False, "priority": "high"}
    ])

    # Filter operators - using >> for transformation
    active_todos = todos >> (lambda todos: [t for t in todos if not t["completed"]])
    completed_todos = todos >> (lambda todos: [t for t in todos if t["completed"]])

    # Count computations
    total_count = todos >> len
    active_count = active_todos >> len
    completed_count = completed_todos >> len

    # Priority filtering
    high_priority_active = active_todos >> (
        lambda todos: [t for t in todos if t["priority"] == "high"]
    )

    # Status indicators
    has_completed_items = completed_count >> (lambda c: c > 0)
    all_completed = active_count >> (lambda c: c == 0)
    is_empty = total_count >> (lambda c: c == 0)

    # Dynamic filtering based on current filter mode
    filter_mode = observable("all")  # "all", "active", "completed"

    filtered_todos = (todos | filter_mode) >> (
        lambda todos, mode: {
            "active": [t for t in todos if not t["completed"]],
            "completed": [t for t in todos if t["completed"]],
            "all": todos
        }.get(mode, todos)
    )

# Usage
store = TodoStore()
print(f"Active: {store.active_count}, Completed: {store.completed_count}")

# Change filter
TodoStore.filter_mode = "active"
print(f"Filtered: {len(store.filtered_todos)} items")

    Reactive Methods¶

    Use @reactive decorators to automatically run methods when dependencies change:

    from fynx import reactive

class NotificationStore(Store):
    user_name = observable("User")
    unread_count = observable(0)
    last_login = observable(None)

    # Reactive method - automatically called when dependencies change
    @staticmethod
    @reactive(user_name, unread_count)
    def update_window_title(name, count):
        if count > 0:
            title = f"({count}) FynX App - {name}"
        else:
            title = f"FynX App - {name}"
        # In a real app, this would update the window title
        print(f"Window title updated: {title}")

    @staticmethod
    @reactive(last_login)
    def on_login_changed(login_time):
        if login_time:
            print(f"User logged in at {login_time}")
        else:
            print("User logged out")

# Usage
NotificationStore.user_name = "Alice"  # Triggers title update
NotificationStore.unread_count = 3     # Triggers title update again
# Output: Window title updated: (3) FynX App - Alice

    Chained Computations¶

    Computed properties can depend on other computed properties:

    class AnalyticsStore(Store):
    raw_data = observable([10, 20, 30, 40, 50])

    # Basic statistics
    count = raw_data >> len
    sum = raw_data >> (lambda data: sum(data))
    mean = (sum | count) >> (lambda total, n: total / n if n > 0 else 0)

    # Advanced statistics that depend on basic ones
    variance = (raw_data | mean | count) >> (
        lambda data, avg, n: (
            sum((x - avg) ** 2 for x in data) / (n - 1)
            if n > 1 else 0
        )
    )

    std_deviation = variance >> (lambda var: var ** 0.5)

    coefficient_of_variation = (std_deviation | mean) >> (
        lambda std, avg: std / avg if avg != 0 else 0
    )

# Usage
analytics = AnalyticsStore()
print(f"Mean: {analytics.mean:.2f}")
print(f"Std Dev: {analytics.std_deviation:.2f}")
print(f"Coeff of Var: {analytics.coefficient_of_variation:.2f}")

    Store Communication¶

    Store-to-Store Dependencies¶

    Stores can depend on state from other stores:

    class UserPreferencesStore(Store):
    theme = observable("light")
    language = observable("en")
    notifications_enabled = observable(True)

class AppStateStore(Store):
    current_user_id = observable("user123")

    # Computed property that depends on another store
    @computed
    def user_theme(self):
        # This creates a dependency on UserPreferencesStore.theme
        return UserPreferencesStore.theme

    @computed
    def display_greeting(self):
        user_id = self.current_user_id
        language = UserPreferencesStore.language

        greetings = {
            "en": f"Hello, {user_id}!",
            "es": f"Hola, {user_id}!",
            "fr": f"Bonjour, {user_id}!"
        }
        return greetings.get(language, greetings["en"])

# Usage
AppStateStore.current_user_id = "alice"
UserPreferencesStore.language = "es"
print(AppStateStore.display_greeting)  # "Hola, alice!"

    Best Practices¶

    Store Organization¶

    1. Single Responsibility: Each store should manage one logical domain
    2. Clear Naming: Use descriptive names for stores and properties
    3. Consistent Patterns: Follow similar patterns across your stores

    State Management¶

    1. Immutability: Treat observables as immutable - create new values rather than modifying existing ones
    2. Validation: Add validation logic in setter methods or computed properties
    3. Error Handling: Handle edge cases gracefully in computed properties

    Performance Considerations¶

    1. Lazy Evaluation: Computed properties only recalculate when accessed
    2. Dependency Tracking: Only relevant observables trigger updates
    3. Avoid Deep Nesting: Keep dependency chains manageable

    Testing Stores¶

    def test_counter_store():
    # Reset store state
    CounterStore.count = 0
    CounterStore.step = 1

    # Test basic operations
    CounterStore.increment()
    assert CounterStore.count.value == 1
    assert CounterStore.doubled_count == 2

    CounterStore.step = 5

    assert CounterStore.doubled_count == 12

    print(" Counter store tests passed")

    Complete Example: Todo Application Store¶

    Here's a comprehensive example adapted from real-world usage:

    from fynx import Store, observable, computed, reactive

class TodoApplicationStore(Store):
    """Complete todo application state management."""

    # Core data
    todos = observable([])
    filter_mode = observable("all")  # "all", "active", "completed"

    # Filtering computed properties
    active_todos = todos >> (lambda todos: [t for t in todos if not t["completed"]])
    completed_todos = todos >> (lambda todos: [t for t in todos if t["completed"]])

    # Statistics
    total_count = todos >> len
    active_count = active_todos >> len
    completed_count = completed_todos >> len

    # Status flags
    is_empty = total_count >> (lambda c: c == 0)
    all_completed = active_count >> (lambda c: c == 0)
    has_completed_items = completed_count >> (lambda c: c > 0)

    # Dynamic filtering
    filtered_todos = (todos | filter_mode) >> (
        lambda todos, mode: {
            "active": [t for t in todos if not t["completed"]],
            "completed": [t for t in todos if t["completed"]],
            "all": todos
        }.get(mode, todos)
    )

    # Human-readable status
    status_text = observable("")

    @staticmethod
    @reactive(is_empty, all_completed, active_count, completed_count)
    def _update_status_text(empty, all_done, active, completed):
        if empty:
            TodoApplicationStore.status_text.set("No todos yet. Add one above!")
        elif all_done:
            TodoApplicationStore.status_text.set(f" All {completed} todos completed!")
        elif completed == 0:
            todo_word = "todo" if active == 1 else "todos"
            TodoApplicationStore.status_text.set(f"{active} active {todo_word} remaining")
        else:
            TodoApplicationStore.status_text.set(f"{active} active, {completed} completed")

    # Public API methods
    @classmethod
    def add_todo(cls, text):
        if text.strip():
            new_todo = {
                "id": str(hash(text + str(len(cls.todos.value)))),
                "text": text.strip(),
                "completed": False
            }
            cls.todos = cls.todos.value + [new_todo]

    @classmethod
    def toggle_todo(cls, todo_id):
        cls.todos = [
            {**todo, "completed": not todo["completed"]}
            if todo["id"] == todo_id else todo
            for todo in cls.todos.value
        ]

    @classmethod
    def delete_todo(cls, todo_id):
        cls.todos = [todo for todo in cls.todos.value if todo["id"] != todo_id]

    @classmethod
    def clear_completed(cls):
        cls.todos = [todo for todo in cls.todos.value if not todo["completed"]]

# Usage example
TodoApplicationStore.add_todo("Learn FynX stores")
TodoApplicationStore.add_todo("Build amazing app")
TodoApplicationStore.add_todo("Write comprehensive tests")

print(f"Status: {TodoApplicationStore.status_text.value}")
# Output: Status: 3 active todos remaining

TodoApplicationStore.toggle_todo(TodoApplicationStore.todos.value[0]["id"])
print(f"Status: {TodoApplicationStore.status_text.value}")
# Output: Status: 2 active, 1 completed

    Stores are incredibly powerful for managing complex application state. They provide the structure and patterns you need to build maintainable, reactive applications with FynX.
